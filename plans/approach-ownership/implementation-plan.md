> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

> ⚠️ **THIS PLAN IS INTENTIONALLY INCOMPLETE.** It covers only **Phase 1**
> (honest connector metric — design Component 1) and **Phase 2** (connector
> features, confidence, evaluation, and the editor labeling + calibration loop —
> design Components 2 & 3). **Phase 3** (design Components 4 & 5 — runtime
> approach ownership with turn-by-turn on the connector, and external-handoff
> demotion) is **NOT planned here** and MUST be planned in a later pass, after
> Phase-2 labeling has produced data and the confidence thresholds have been
> calibrated. Do not attempt Phase 3 from this document. See
> `plans/approach-ownership/design.md`.

**Goal:** Ship the editor data + confidence tooling: an honest connector-quality metric that separates unreachable origins from real routing gaps, a shared pure feature/confidence/evaluation core, and an editor labeling workflow + calibration loop that turns human valid/unacceptable verdicts into interpretable, data-tuned confidence thresholds.

**Architecture:** Pure, node-tested core modules in `@cycleways/core` (`connectorFeatures`, `connectorConfidence`, `connectorEvaluate`) reused identically by the editor now and by the app in Phase 3. The editor labels connector routes into an append-only JSONL dataset via new server endpoints, and a calibration panel scores the current thresholds against that dataset client-side (same shared modules). Connector features are derived from `previewBaseRoute`'s existing `edgeCosts` diagnostics — no routing-engine change.

**Tech Stack:** Node ESM (`.mjs` tests run with `node tests/<file>.mjs`), the `@cycleways/core` workspace package, the editor's plain-Node `http` server (`editor/server.mjs`) and vanilla-JS client (`editor/editor.js`) with Mapbox GL layers, `node:crypto` for strategy hashing.

## Global Constraints

- **No map-data edits.** Never write `data/map-source.geojson`, `public-data/`, or any base-graph asset. The label dataset is a *new, separate* artifact.
- **`data/connector-eval/` is non-pipeline.** It must never enter the Build→Promote flow. Its files (`labels.jsonl`, `strategies.json`) are **committed to git** as versioned ground truth — but only via these editor endpoints, never regenerated by Build.
- **Shared feature/confidence logic is single-source.** `connectorFeatures` / `connectorConfidence` / `connectorEvaluate` live in `@cycleways/core` and are the only implementations; the editor imports them (Phase 3's app runtime will import the same modules). No reimplementation in the editor.
- **Features come from `previewBaseRoute().edgeCosts`.** Each traversal diagnostic already carries `routeClass`, `roadType`, `cyclewaysSegmentIds`, and per-traversal `distanceMeters`. Do not add a new per-edge descriptor to `route-manager.js`.
- **Confidence is interpretable thresholds, not ML.** `DEFAULT_CONNECTOR_THRESHOLDS` ships as working, calibrated-later defaults (concrete numbers, tuned from labels between Phase 2 and Phase 3).
- **Tests are added to the `test` npm script** (the long `&&` chain in `package.json`). Run an individual test with `node tests/<file>.mjs`.
- **CW membership** is detected by `hasCyclewaysNetworkMembership(edge)` (already in `connectorCostModel.js`): truthy `cwSegmentIds`/`cyclewaysSegmentIds`. In `edgeCosts` diagnostics the field is `cyclewaysSegmentIds`.

```js
import { getDistance } from "../utils/distance.js";

// Worst-to-best ordering of connector edge classes for a "how sketchy is this
// path" signal. cw-owned edges are treated as the best class (cw_network).
export const CONNECTOR_CLASS_RANK = {
  cw_network: 0,
  road: 1,
  local_road: 2,
  cycle: 3,
  path_track: 4,
  manual: 5,
  other: 6,
};

function edgeClass(edge) {
  const cwOwned =
    Array.isArray(edge.cyclewaysSegmentIds) && edge.cyclewaysSegmentIds.length > 0;
  if (cwOwned) return "cw_network";
  if (edge.routeClass === "road" || edge.roadType === "road") return "road";
  const rc = edge.routeClass;
  return rc != null && rc in CONNECTOR_CLASS_RANK ? rc : "other";
}

export function computeConnectorFeatures(preview, { origin, routeStart } = {}) {
  const straightLineMeters =
    origin && routeStart ? getDistance(origin, routeStart) : 0;
  if (!preview || preview.failure) {
    return {
      snapOk: false,
      failure: preview ? preview.failure : "no-preview",
      straightLineMeters,
      routedMeters: 0,
      detourRatio: Infinity,
      cwNetworkFraction: 0,
      worstRouteClass: null,
      edgeCount: 0,
    };
  }
  const edges = Array.isArray(preview.edgeCosts) ? preview.edgeCosts : [];
  const routedMeters = Number(preview.distanceMeters) || 0;
  let cwMeters = 0;
  let worstRank = -1;
  let worstClass = null;
  for (const edge of edges) {
    const cls = edgeClass(edge);
    const d = Number(edge.distanceMeters) || 0;
    if (cls === "cw_network") cwMeters += d;
    const rank = CONNECTOR_CLASS_RANK[cls] ?? CONNECTOR_CLASS_RANK.other;
    if (rank > worstRank) {
      worstRank = rank;
      worstClass = cls;
    }
  }
  return {
    snapOk: true,
    failure: null,
    straightLineMeters,
    routedMeters,
    detourRatio:
      straightLineMeters > 1
        ? routedMeters / straightLineMeters
        : routedMeters > 0
        ? Infinity
        : 1,
    cwNetworkFraction: routedMeters > 0 ? cwMeters / routedMeters : 0,
    worstRouteClass: worstClass,
    edgeCount: edges.length,
  };
}
```

```js
import { CONNECTOR_CLASS_RANK } from "./connectorFeatures.js";

// Interpretable, hand-tunable thresholds. Values are calibrated-later defaults:
// set from the editor's labeled dataset (design Component 3) before Phase 3.
export const DEFAULT_CONNECTOR_THRESHOLDS = {
  guideRadiusMeters: 3000,
  tooFarRadiusMeters: 15000,
  maxDetourRatio: 2.5,
  maxRoutedMeters: 8000,
  worstClassAllowed: "local_road",
};

export function classifyConnector(features, thresholds = DEFAULT_CONNECTOR_THRESHOLDS) {
  if (!features || !features.snapOk) {
    return { tier: "too-far", handoffSuggested: true, reasons: ["unreachable"] };
  }
  if (features.straightLineMeters > thresholds.tooFarRadiusMeters) {
    return { tier: "too-far", handoffSuggested: true, reasons: ["beyond-too-far-radius"] };
  }
  const allowedRank =
    CONNECTOR_CLASS_RANK[thresholds.worstClassAllowed] ?? CONNECTOR_CLASS_RANK.other;
  const worstRank =
    features.worstRouteClass != null
      ? CONNECTOR_CLASS_RANK[features.worstRouteClass] ?? CONNECTOR_CLASS_RANK.other
      : 0;
  const reasons = [];
  if (features.straightLineMeters > thresholds.guideRadiusMeters) reasons.push("beyond-guide-radius");
  if (features.detourRatio > thresholds.maxDetourRatio) reasons.push("detour-too-high");
  if (features.routedMeters > thresholds.maxRoutedMeters) reasons.push("routed-too-long");
  if (worstRank > allowedRank) reasons.push("class-too-low");
  if (reasons.length === 0) {
    return { tier: "guide", handoffSuggested: false, reasons: [] };
  }
  return { tier: "show-leg", handoffSuggested: true, reasons };
}
```

```js
import { classifyConnector } from "./connectorConfidence.js";

// Score a set of human-labeled connectors against a threshold set. A "valid"
// label should be guided; an "unacceptable" label should not; "borderline" is
// counted for visibility but excluded from the precision/recall rates.
export function evaluateThresholds(labels, thresholds) {
  const counts = {
    valid: { guide: 0, other: 0 },
    unacceptable: { guide: 0, other: 0 },
    borderline: { guide: 0, other: 0 },
    total: 0,
  };
  for (const rec of labels || []) {
    if (!rec || !counts[rec.verdict]) continue;
    const { tier } = classifyConnector(rec.features, thresholds);
    counts[rec.verdict][tier === "guide" ? "guide" : "other"] += 1;
    counts.total += 1;
  }
  const validTotal = counts.valid.guide + counts.valid.other;
  const invalidTotal = counts.unacceptable.guide + counts.unacceptable.other;
  return {
    counts,
    validGuideRate: validTotal ? counts.valid.guide / validTotal : null,
    invalidGuideRate: invalidTotal ? counts.unacceptable.guide / invalidTotal : null,
  };
}
```

```js
import { appendFile, readFile, writeFile, mkdir } from "node:fs/promises";
import { dirname } from "node:path";
import { createHash } from "node:crypto";

const VERDICTS = new Set(["valid", "unacceptable", "borderline"]);

function isLatLng(p) {
  return p && Number.isFinite(Number(p.lat)) && Number.isFinite(Number(p.lng));
}

export function validateLabelRecord(rec) {
  if (!rec || typeof rec !== "object") return "record object required";
  if (!isLatLng(rec.routeStart)) return "routeStart {lat,lng} required";
  if (!isLatLng(rec.origin)) return "origin {lat,lng} required";
  if (!VERDICTS.has(rec.verdict)) return "verdict must be valid|unacceptable|borderline";
  if (!rec.features || typeof rec.features !== "object") return "features object required";
  return null;
}

// Deterministic JSON with sorted keys so a hash is independent of key order.
function stableStringify(value) {
  if (value === null || typeof value !== "object") return JSON.stringify(value);
  if (Array.isArray(value)) return `[${value.map(stableStringify).join(",")}]`;
  const keys = Object.keys(value).sort();
  return `{${keys.map((k) => `${JSON.stringify(k)}:${stableStringify(value[k])}`).join(",")}}`;
}

export function hashStrategy(strategy) {
  const hex = createHash("sha256").update(stableStringify(strategy)).digest("hex");
  return `sha256:${hex}`;
}

export async function appendLabel(path, rec) {
  const error = validateLabelRecord(rec);
  if (error) {
    const err = new Error(error);
    err.status = 400;
    throw err;
  }
  const record = { ...rec, ts: new Date().toISOString() };
  await mkdir(dirname(path), { recursive: true });
  await appendFile(path, `${JSON.stringify(record)}\n`, "utf8");
  return record;
}

export async function readLabels(path) {
  let text;
  try {
    text = await readFile(path, "utf8");
  } catch (e) {
    if (e.code === "ENOENT") return [];
    throw e;
  }
  return text.split("\n").filter(Boolean).map((line) => JSON.parse(line));
}

export async function upsertStrategy(path, strategy) {
  const hash = hashStrategy(strategy);
  let existing = {};
  try {
    existing = JSON.parse(await readFile(path, "utf8"));
  } catch (e) {
    if (e.code !== "ENOENT") throw e;
  }
  if (!existing[hash]) {
    existing[hash] = strategy;
    await mkdir(dirname(path), { recursive: true });
    await writeFile(path, `${JSON.stringify(existing, null, 2)}\n`, "utf8");
  }
  return hash;
}
```
